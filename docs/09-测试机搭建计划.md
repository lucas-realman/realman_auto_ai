**Step 13: 云端多模型 Judge 评估器骨架**

```python
# tests/eval/judge.py — Agent 对话质量云端多模型评估器

"""
云端多模型 Judge: 用 GPT-4.1 / Qwen-Max / DeepSeek-V3 三模型交叉评估 Agent 的对话质量。
不再使用本地 vLLM 做 Judge，避免“学生给自己批作业”。 

vLLM 专用于 Agent 集成测试推理，Judge 评估全部走云端 API。

评估维度:
1. 意图理解 (1-5): Agent 是否正确识别了用户意图？
2. 操作准确 (1-5): Tool 调用参数是否正确？
3. 回复质量 (1-5): 回复是否清晰、完整、格式正确？
4. 异常处理 (1-5): 异常场景下是否有合理应对？

加权总分 = GPT-4.1×0.40 + Qwen-Max×0.35 + DeepSeek-V3×0.25
≥ 4.0 为绿灯，3.0-4.0 为黄灯，< 3.0 为红灯。
"""

import json
import asyncio
from openai import AsyncOpenAI

# 云端 Judge 模型配置 (详见 05-§4.2)
JUDGE_MODELS = [
    {
        "name": "GPT-4.1",
        "client": AsyncOpenAI(base_url="https://api.openai.com/v1"),
        "model": "gpt-4.1",
        "weight": 0.40,
    },
    {
        "name": "Qwen-Max",
        "client": AsyncOpenAI(base_url="https://dashscope.aliyuncs.com/compatible-mode/v1"),
        "model": "qwen-max",
        "weight": 0.35,
    },
    {
        "name": "DeepSeek-V3",
        "client": AsyncOpenAI(base_url="https://api.deepseek.com/v1"),
        "model": "deepseek-chat",
        "weight": 0.25,
    },
]

JUDGE_PROMPT = """你是一个 AI 系统评估专家。请评估以下 CRM Agent 的对话质量。

## 用户输入
{user_input}

## Agent 回复
{agent_response}

## 预期行为
{expected_behavior}

请按以下维度打分 (1-5 分):
1. 意图理解: Agent 是否正确识别了用户意图？
2. 操作准确: 执行的操作是否正确？参数是否准确？
3. 回复质量: 回复是否清晰、完整、专业？
4. 异常处理: 对于异常情况是否有合理应对？

请严格按以下 JSON 格式输出（不要添加其他内容）:
{{"intent_score": X, "action_score": X, "reply_score": X, "error_handling_score": X, "comment": "简短评语"}}
"""


async def _call_single_judge(judge_cfg: dict, prompt: str) -> dict:
    """调用单个云端 Judge 模型。"""
    resp = await judge_cfg["client"].chat.completions.create(
        model=judge_cfg["model"],
        messages=[{"role": "user", "content": prompt}],
        max_tokens=200,
        temperature=0.1,
    )
    result = resp.choices[0].message.content
    scores = json.loads(result)
    scores["judge"] = judge_cfg["name"]
    scores["weight"] = judge_cfg["weight"]
    scores["avg_score"] = round(
        (scores["intent_score"] + scores["action_score"] +
         scores["reply_score"] + scores["error_handling_score"]) / 4, 2
    )
    return scores


async def judge_single(user_input: str, agent_response: str,
                       expected_behavior: str) -> dict:
    """对单条 Agent 对话进行多模型交叉评估。"""
    prompt = JUDGE_PROMPT.format(
        user_input=user_input,
        agent_response=agent_response,
        expected_behavior=expected_behavior,
    )
    # 并行调用 3 个云端 Judge
    tasks = [_call_single_judge(j, prompt) for j in JUDGE_MODELS]
    results = await asyncio.gather(*tasks, return_exceptions=True)

    valid = [r for r in results if isinstance(r, dict)]
    # 加权计算总分
    weighted_score = sum(r["avg_score"] * r["weight"] for r in valid)
    total_weight = sum(r["weight"] for r in valid)

    return {
        "weighted_avg_score": round(weighted_score / total_weight, 2) if total_weight else 0,
        "per_judge": valid,
        "divergence": max(r["avg_score"] for r in valid) - min(r["avg_score"] for r in valid) if valid else 0,
    }


def traffic_light(avg_score: float) -> str:
    """根据加权平均分返回红绿灯。"""
    if avg_score >= 4.0:
        return "🟢 GREEN"
    elif avg_score >= 3.0:
        return "🟡 YELLOW"
    else:
        return "🔴 RED"
```

**Step 14: 报告生成器骨架**

```python
# tests/report/generate_report.py — 综合报告生成

"""
将 pytest + Agent评估 + 性能基线 汇总为一份红/绿/黄报告。

报告格式 (summary_YYYYMMDD_HHMMSS.json):
{
  "timestamp": "2026-03-10T11:00:00",
  "overall": "GREEN",
  "pytest": {"total": 50, "passed": 48, "failed": 2, "status": "YELLOW"},
  "eval": {"avg_score": 4.2, "cases": 20, "status": "GREEN"},
  "perf": {"ttft_p95": 1.8, "throughput": 25.3, "status": "GREEN"},
  "details": [...]
}
"""

import json
import sys
import xml.etree.ElementTree as ET
from datetime import datetime


def parse_junit(junit_path: str) -> dict:
    """解析 pytest JUnit XML 报告。"""
    tree = ET.parse(junit_path)
    root = tree.getroot()
    suite = root.find("testsuite") or root
    return {
        "total": int(suite.get("tests", 0)),
        "passed": int(suite.get("tests", 0)) - int(suite.get("failures", 0)) - int(suite.get("errors", 0)),
        "failed": int(suite.get("failures", 0)) + int(suite.get("errors", 0)),
        "status": "GREEN" if int(suite.get("failures", 0)) == 0 else "YELLOW",
    }


def parse_eval(eval_path: str) -> dict:
    """解析 Agent 评估结果。"""
    with open(eval_path) as f:
        data = json.load(f)
    avg = data.get("avg_score", 0)
    return {
        "avg_score": avg,
        "cases": data.get("total_cases", 0),
        "status": "GREEN" if avg >= 4.0 else ("YELLOW" if avg >= 3.0 else "RED"),
    }


def parse_perf(perf_path: str) -> dict:
    """解析性能基线。"""
    with open(perf_path) as f:
        data = json.load(f)
    ttft = data.get("ttft_p95", 999)
    return {
        "ttft_p95": ttft,
        "throughput": data.get("throughput_tps", 0),
        "status": "GREEN" if ttft < 3.0 else ("YELLOW" if ttft < 5.0 else "RED"),
    }


def generate(junit_path, eval_path, perf_path, output_path):
    """生成综合报告。"""
    pytest_result = parse_junit(junit_path)
    eval_result = parse_eval(eval_path)
    perf_result = parse_perf(perf_path)

    # 总体状态: 取最差的
    statuses = [pytest_result["status"], eval_result["status"], perf_result["status"]]
    if "RED" in statuses:
        overall = "RED"
    elif "YELLOW" in statuses:
        overall = "YELLOW"
    else:
        overall = "GREEN"

    report = {
        "timestamp": datetime.now().isoformat(),
        "overall": overall,
        "pytest": pytest_result,
        "eval": eval_result,
        "perf": perf_result,
    }

    with open(output_path, "w") as f:
        json.dump(report, f, indent=2, ensure_ascii=False)

    # 控制台输出
    light = {"GREEN": "🟢", "YELLOW": "🟡", "RED": "🔴"}
    print(f"\n{'='*50}")
    print(f"  {light[overall]} 总体状态: {overall}")
    print(f"  pytest:  {light[pytest_result['status']]} {pytest_result['passed']}/{pytest_result['total']} passed")
    print(f"  Agent:   {light[eval_result['status']]} avg={eval_result['avg_score']}/5.0 ({eval_result['cases']} cases)")
    print(f"  性能:    {light[perf_result['status']]} TTFT P95={perf_result['ttft_p95']}s")
    print(f"{'='*50}\n")


if __name__ == "__main__":
    import argparse
    parser = argparse.ArgumentParser()
    parser.add_argument("--pytest", required=True)
    parser.add_argument("--eval", required=True)
    parser.add_argument("--perf", required=True)
    parser.add_argument("--output", required=True)
    args = parser.parse_args()
    generate(args.pytest, args.eval, args.perf, args.output)
```
